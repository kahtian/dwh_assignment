DROP TABLE supplier;

SET LINESIZE 200;
SET PAGESIZE 200;
SET DEFINE OFF;


COLUMN supplierID         FORMAT A8;
COLUMN supplierName       FORMAT A30;
COLUMN supplierLocation   FORMAT A15;
COLUMN supplierContactNum FORMAT A15;
COLUMN supplierEmail      FORMAT A30;

CREATE TABLE Supplier (
    supplierID         VARCHAR2(5) NOT NULL,
    supplierName       VARCHAR2(100) NOT NULL,
    supplierLocation   VARCHAR2(100) NOT NULL,
    supplierContactNum VARCHAR2(20) NOT NULL,
    supplierEmail      VARCHAR2(100) NOT NULL,
    PRIMARY KEY (supplierID),
    CONSTRAINT chk_supplierID_format CHECK (REGEXP_LIKE(supplierID, '^SP[0-9]{2}$'))
);
